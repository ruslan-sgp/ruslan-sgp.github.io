# ==================================================
# Задача 1
# ==================================================

Pairs = {
    6: "принтер",
    11: "плоттер",
    3: "сканер",
    7: "монитор",
    15: "микрофон"
}

print("Задача 1")
print("Значения где ключ однозначное число:")
for key, value in Pairs.items():
    if 0 <= key <= 9:
        print(f"{key} — {value}")


# ==================================================
# Задача 2
# ==================================================

M = {
    6: "принтер",
    11: "плоттер",
    3: "сканер",
    7: "монитор",
    15: "микрофон"
}

# Получаем список ключей
keys = list(M.keys())

# Первый и последний ключ
first_key = keys[0]
last_key = keys[-1]

# Меняем местами значения по этим ключам
M[first_key], M[last_key] = M[last_key], M[first_key]

print("\nЗадача 2")
print("Обновленный словарь:")
for key, value in M.items():
    print(f"{key}: {value}")

# Создаём словарь
M = {
    10: [0, 1, 2, 3, 4],
    15: [1, 2, 3, 4, 5, 6],
    12: [10, 11, 12],
    17: [5, 2]
}


print("\nЗадача 3")

# Вывод всех ключей
print("Ключи словаря:")
for key in M.keys():
    print(key)

# Вывод всех значений
print("Значения словаря:")
for value in M.values():
    print(value)

# Создаём словарь с кодами работ и оценками
RD = {
    "P24": "хорошо",
    "D3": "отлично",
    "P21": "хорошо",
    "P18": "удовлетворительно",
    "P9": "удовлетворительно",
    "D7": "хорошо",
    "D11": "хорошо"
}

print("\nЗадача 4")
print("Работы с оценками 'хорошо' и 'отлично':")

# Фильтруем и выводим только те работы, которые оценены на 'хорошо' или 'отлично'
for code, grade in RD.items():
    if grade in ("хорошо", "отлично"):
        print(f"{code}: {grade}")
